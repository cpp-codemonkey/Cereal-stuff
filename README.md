  FColor
  FDateTime
  FIntPoint

  FIntVector
  FLinearColor
  FMatrix
  FMatrix2x2
  FName
  FOrientedBox
  FPlane
  FQuat
  FQuat2D
  FRotator
  FScale
  FScale2D
  FShear2D
  FSphere
  FString
  FText
  FTimespan
  FTransform
  FTwoVector
  FVector
  FVector2D
  TArray
  TBigInt
  TInterval
  TMap
  TSet
  TSubclassOf<AActor>
